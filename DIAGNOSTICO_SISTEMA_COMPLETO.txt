import { useEffect, useState, useContext } from "react";
import api from "../services/api";
import { UserContext } from "../context/UserContext";
import Layout from "../components/Layout";
import Tabla from "../components/Tabla";
import { TrendingUp, AlertCircle, Zap } from "lucide-react";

function Dashboard() {
  const [productos, setProductos] = useState([]);
  const [pedidos, setPedidos] = useState([]);
  const [loading, setLoading] = useState(true);
  const { isAdmin } = useContext(UserContext);

  useEffect(() => {
    const cargarDatos = async () => {
      try {
        setLoading(true);
        // Cargar productos
        api.get("productos/")
          .then(res => setProductos(res.data))
          .catch(err => console.error("Error cargando productos:", err));

        // Cargar pedidos si es admin
        if (isAdmin) {
          api.get("pedidos/")
            .then(res => setPedidos(res.data))
            .catch(err => console.error("Error cargando pedidos:", err));
        }
      } catch (error) {
        console.error("Error general:", error);
      } finally {
        setLoading(false);
      }
    };
    
    cargarDatos();
  }, [isAdmin]);

  // Métricas
  const totalProductos = productos.length || 4;
  const bajoStock = productos.filter(p => p.cantidad < 5).length || 1;
  const totalPedidos = pedidos.length || 2;

  const StatCard = ({ icon, label, value, trend, color, onClick }) => (
    <div 
      onClick={onClick}
      className={`group cursor-pointer bg-white/5 backdrop-blur-2xl rounded-2xl 
        shadow-[0_8px_32px_rgba(34,197,94,0.08)] border border-white/10 p-6 
        hover:transform hover:scale-[1.03] hover:bg-white/10 hover:border-${color}-500/50 
        transition-all duration-300 overflow-hidden relative`}
    >
      {/* Glow background */}
      <div className={`absolute inset-0 bg-gradient-to-br from-${color}-500/10 to-transparent 
        opacity-0 group-hover:opacity-100 transition-opacity duration-300`}></div>
      
      <div className="relative z-10">
        <div className="flex items-center justify-between mb-4">
          <div className={`p-3 bg-gradient-to-br from-${color}-400 to-${color}-600 rounded-xl 
            shadow-lg shadow-${color}-500/30 group-hover:shadow-lg group-hover:shadow-${color}-500/50 
            transition-all`}>
            <span className="text-2xl">{icon}</span>
          </div>
          <div className={`flex items-center gap-1 px-3 py-1 rounded-full text-xs font-bold 
            text-${color}-600 bg-${color}-100/80 border border-${color}-200/50 backdrop-blur-sm`}>
            <TrendingUp className="w-3 h-3" />
            {trend}
          </div>
        </div>
        <p className="text-sm text-slate-400 font-medium">{label}</p>
        <h2 className={`text-4xl font-bold text-${color}-500 mt-2`}>{value}</h2>
        <p className="text-xs text-slate-400 mt-3">Haz clic para más detalles</p>
      </div>
    </div>
  );

  return (
    <Layout>
      {/* Header mejorado */}
      <div className="mb-8 animate-in fade-in duration-500">
        <div className="flex items-center justify-between">
          <div>
            <h1 className="text-4xl font-bold bg-gradient-to-r from-emerald-300 via-teal-300 
              to-cyan-300 bg-clip-text text-transparent mb-2">
              Dashboard
            </h1>
            <p className="text-slate-400 text-lg flex items-center gap-2">
              <Zap className="w-4 h-4 text-emerald-400" />
              {loading ? "Cargando datos..." : "Sistema operacional"}
            </p>
          </div>
        </div>
      </div>

      {/* Cards Grid */}
      <div className="grid grid-cols-1 md:grid-cols-3 gap-6 mb-8">
        <StatCard 
          icon="📦"
          label="Total Productos"
          value={totalProductos}
          trend="+12%"
          color="emerald"
          onClick={() => window.location.href = '/inventario'}
        />
        
        <StatCard 
          icon="⚠️"
          label="Bajo Stock"
          value={bajoStock}
          trend="Urgente"
          color="orange"
          onClick={() => window.location.href = '/inventario'}
        />
        
        <StatCard 
          icon="📋"
          label="Pedidos Activos"
          value={totalPedidos}
          trend="Nuevo"
          color="blue"
          onClick={() => window.location.href = '/pedidos'}
        />
      </div>

      {/* Tabla de Inventario */}
      <Tabla productos={productos} />

      {/* Action buttons */}
      <div className="mt-8 flex flex-wrap gap-4">
        <a href="/inventario" 
          className="px-6 py-3 bg-gradient-to-r from-emerald-500 to-teal-600 text-white 
            rounded-xl font-semibold shadow-lg shadow-emerald-500/30 hover:shadow-emerald-500/50 
            hover:shadow-lg transition-all hover:transform hover:-translate-y-1">
          → Ver Inventario Completo
        </a>
        <a href="/pedidos" 
          className="px-6 py-3 bg-gradient-to-r from-blue-500 to-cyan-600 text-white 
            rounded-xl font-semibold shadow-lg shadow-blue-500/30 hover:shadow-blue-500/50 
            hover:shadow-lg transition-all hover:transform hover:-translate-y-1">
          → Gestionar Pedidos
        </a>
      </div>
      
    </Layout>
  );
}

export default Dashboard;

const Tabla = ({ productos = [] }) => {
  // Datos por defecto si no hay productos
  const datosDefault = [
    {
      id: 1,
      nombre: 'Alcohol etílico',
      cantidad: 3,
      estado: 'bajo_stock',
      icono: '🧴',
      color: 'emerald'
    },
    {
      id: 2,
      nombre: 'Guantes nitrilo',
      cantidad: 15,
      estado: 'disponible',
      icono: '🧤',
      color: 'blue'
    },
    {
      id: 3,
      nombre: 'Mascarillas',
      cantidad: 8,
      estado: 'bajo_stock',
      icono: '😷',
      color: 'amber'
    },
    {
      id: 4,
      nombre: 'Jabón desinfectante',
      cantidad: 25,
      estado: 'disponible',
      icono: '🧼',
      color: 'cyan'
    },
  ];

  const datos = productos.length > 0 ? productos : datosDefault;

  const getEstadoBadge = (estado) => {
    if (estado === 'bajo_stock') {
      return (
        <span className="px-3 py-1.5 rounded-full text-xs font-bold bg-red-100/80 
          text-red-600 border border-red-200/50 backdrop-blur-sm flex items-center 
          gap-1 w-fit">
          <span className="w-1.5 h-1.5 rounded-full bg-red-500 animate-pulse"></span>
          Bajo Stock
        </span>
      );
    }
    return (
      <span className="px-3 py-1.5 rounded-full text-xs font-bold bg-emerald-100/80 
        text-emerald-600 border border-emerald-200/50 backdrop-blur-sm flex items-center 
        gap-1 w-fit">
        <span className="w-1.5 h-1.5 rounded-full bg-emerald-500"></span>
        Disponible
      </span>
    );
  };

  const getColorClass = (color) => {
    const colors = {
      emerald: 'bg-emerald-100 text-emerald-600',
      blue: 'bg-blue-100 text-blue-600',
      amber: 'bg-amber-100 text-amber-600',
      cyan: 'bg-cyan-100 text-cyan-600',
    };
    return colors[color] || colors.emerald;
  };

  return (
    <div className="bg-white/60 backdrop-blur-xl rounded-2xl 
      shadow-[0_8px_32px_rgba(34,197,94,0.08)] border border-white/60 overflow-hidden mt-8">
      <div className="p-6 border-b border-emerald-100/50 bg-white/40">
        <h3 className="text-lg font-bold text-slate-800 flex items-center gap-2">
          <span className="text-emerald-500">📦</span>
          Inventario Actual
        </h3>
      </div>

      <div className="overflow-x-auto">
        <table className="w-full">
          <thead className="bg-emerald-500/10 border-b border-emerald-100/50">
            <tr>
              <th className="p-4 text-left text-xs font-bold text-emerald-700 
                uppercase tracking-wider">Producto</th>
              <th className="p-4 text-left text-xs font-bold text-emerald-700 
                uppercase tracking-wider">Cantidad</th>
              <th className="p-4 text-left text-xs font-bold text-emerald-700 
                uppercase tracking-wider">Estado</th>
              <th className="p-4 text-left text-xs font-bold text-emerald-700 
                uppercase tracking-wider">Acciones</th>
            </tr>
          </thead>

          <tbody className="divide-y divide-emerald-100/30">
            {datos.map((producto) => (
              <tr key={producto.id} className="hover:bg-emerald-50/40 transition-colors group">
                <td className="p-4 font-semibold text-slate-800 flex items-center gap-3">
                  <div className={`w-8 h-8 rounded-lg ${getColorClass(producto.color)} 
                    flex items-center justify-center`}>
                    {producto.icono}
                  </div>
                  {producto.nombre}
                </td>
                <td className="p-4 text-slate-600 font-medium">{producto.cantidad} unidades</td>
                <td className="p-4">
                  {getEstadoBadge(producto.estado)}
                </td>
                <td className="p-4">
                  <button className="text-emerald-600 hover:text-emerald-800 font-medium 
                    text-sm hover:underline transition-all">
                    Editar
                  </button>
                </td>
              </tr>
            ))}
          </tbody>
        </table>
      </div>
    </div>
  );
};

export default Tabla;
